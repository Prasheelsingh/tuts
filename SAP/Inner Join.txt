             lips~posnr
             lips~werks
             marc~matnr

          FROM marc INNER JOIN lips
          ON lips~matnr = marc~matnr AND
             lips~werks = marc~werks
          INTO TABLE gt_marc
          FOR ALL ENTRIES IN gt_temp
          WHERE lips~vbeln = gt_temp-delnum.
